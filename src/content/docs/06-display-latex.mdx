import AttoDemo from '../../components/AttoDemo.astro';

{/* Aligned LaTeX needs literal 2-space indentation inside the source
    `<pre>` to align the `&` columns visually. When the source is
    inlined as a JSX template-literal expression (`set:html={\`…\`}`),
    something in the MDX → JS pipeline normalizes leading whitespace
    on continuation lines and the indent disappears. Stashing the
    string in a top-level `export const` bypasses that path: it's a
    plain JS module-scoped string by the time set:html sees it. */}
export const alignedSource = `<code>The late-fee calculation, written out:

<span class="tok-math">$$
\\begin&#123;aligned&#125;
L &amp;= B \\times r \\\\
  &amp;= B \\times 0.015 \\\\
  &amp;= 0.015 B \\quad \\text&#123;(per day)&#125;
\\end&#123;aligned&#125;
$$</span></code>`;

Wrap an expression in double dollar signs: `$$…$$` on their own
lines. Renders as a centered display block, isolated from
surrounding prose.
